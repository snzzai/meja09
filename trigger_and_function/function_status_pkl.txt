CREATE FUNCTION ketStatusPKL(status BOOLEAN) RETURNS VARCHAR(50)
DETERMINISTIC
BEGIN
    IF status = 0 THEN
        RETURN 'Belum diterima PKL';
    ELSEIF status = 1 THEN
        RETURN 'Sudah diterima PKL';
    ELSE
        RETURN 'Status tidak diketahui';
    END IF;
END //
